### 1.2 技术选型理由

#### 前端技术栈

| 技术 | 选型 | 理由 |
|------|------|------|
| 前端框架 | Vue 3 | Composition API、更好的TypeScript支持、性能优化 |
| UI组件库 | Element Plus | 完整的企业级组件、中文文档友好、定制性强 |
| 状态管理 | Pinia | Vue 3官方推荐、API简洁、TypeScript支持好 |
| 构建工具 | Vite | 极速热更新、基于ESM、开发体验好 |
| HTTP库 | Axios | 拦截器强大、请求取消、并发控制 |
| 图表库 | ECharts | 功能强大、配置灵活、免费商用 |

#### 后端技术栈

| 技术 | 选型 | 理由 |
|------|------|------|
| 后端框架 | Spring Boot 2.x | 企业级标准、生态完善、快速开发 |
| JDK版本 | Java 1.8 | 稳定版本、广泛兼容、长期支持 |
| ORM框架 | MyBatis-Plus | 基于MyBatis增强、代码生成、CRUD简化 |
| 数据库 | MySQL 5.7+ | 开源可靠、事务支持、索引优化 |
| 缓存 | Redis 6.x | 高性能、数据结构丰富、支持分布式 |
| JWT | jjwt | 无状态认证、跨域支持、移动端友好 |
| 文档工具 | Knife4j | Swagger增强、界面美观、调试方便 |
| 工具库 | Lombok | 减少样板代码、提高开发效率 |
| 构建工具 | Maven 3.6+ | 依赖管理、项目标准化、插件丰富 |

```java
// common/exception/GlobalExceptionHandler.java
package com.pindou.timer.common.exception;

import com.pindou.timer.common.enums.ErrorCode;
import com.pindou.timer.common.result.Result;
import lombok.extern.slf4j.Slf4j;
import org.springframework.validation.BindException;
import org.springframework.validation.FieldError;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;

import javax.servlet.http.HttpServletRequest;
import javax.validation.ConstraintViolation;
import javax.validation.ConstraintViolationException;

/**
 * 全局异常处理器
 */
@Slf4j
@RestControllerAdvice
public class GlobalExceptionHandler {

    /**
     * 业务异常
     */
    @ExceptionHandler(BusinessException.class)
    public Result<?> handleBusinessException(BusinessException e, HttpServletRequest request) {
        log.error("业务异常: uri={}, message={}", request.getRequestURI(), e.getMessage());
        return Result.error(e.getCode(), e.getMessage());
    }

    /**
     * 参数校验异常（RequestBody）
     */
    @ExceptionHandler(MethodArgumentNotValidException.class)
    public Result<?> handleValidException(MethodArgumentNotValidException e) {
        FieldError fieldError = e.getBindingResult().getFieldError();
        String message = fieldError != null ? fieldError.getDefaultMessage() : "参数校验失败";
        return Result.error(400, message);
    }

    /**
     * 参数绑定异常
     */
    @ExceptionHandler(BindException.class)
    public Result<?> handleBindException(BindException e) {
        FieldError fieldError = e.getFieldError();
        String message = fieldError != null ? fieldError.getDefaultMessage() : "参数绑定失败";
        return Result.error(400, message);
    }

    /**
     * 参数校验异常（RequestParam）
     */
    @ExceptionHandler(ConstraintViolationException.class)
    public Result<?> handleConstraintViolationException(ConstraintViolationException e) {
        String message = e.getConstraintViolations().stream()
                .map(ConstraintViolation::getMessage)
                .findFirst()
                .orElse("参数校验失败");
        return Result.error(400, message);
    }

    /**
     * 系统异常
     */
    @ExceptionHandler(Exception.class)
    public Result<?> handleException(Exception e, HttpServletRequest request) {
        log.error("系统异常: uri={}, message={}", request.getRequestURI(), e.getMessage(), e);
        return Result.error(500, "系统繁忙，请稍后重试");
    }
}
```

### 2.8 JWT工具类

```java
// common/util/JwtUtil.java
package com.pindou.timer.common.util;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.security.Keys;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import javax.crypto.SecretKey;
import java.nio.charset.StandardCharsets;
import java.util.Date;
import java.util.HashMap;
import java.util.Map;

/**
 * JWT工具类
 */
@Component
public class JwtUtil {

    @Value("${jwt.secret}")
    private String secret;

    @Value("${jwt.expiration}")
    private Long expiration;

    /**
     * 生成密钥
     */
    private SecretKey getSignKey() {
        return Keys.hmacShaKeyFor(secret.getBytes(StandardCharsets.UTF_8));
    }

    /**
     * 生成Token
     */
    public String generateToken(String userId, String username) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("userId", userId);
        claims.put("username", username);
        return createToken(claims, username);
    }

    /**
     * 创建Token
     */
    private String createToken(Map<String, Object> claims, String subject) {
        Date now = new Date();
        Date expiryDate = new Date(now.getTime() + expiration);

        return Jwts.builder()
                .setClaims(claims)
                .setSubject(subject)
                .setIssuedAt(now)
                .setExpiration(expiryDate)
                .signWith(getSignKey())
                .compact();
    }

    /**
     * 从Token中获取用户名
     */
    public String getUsernameFromToken(String token) {
        return getClaimsFromToken(token).getSubject();
    }

    /**
     * 从Token中获取用户ID
     */
    public String getUserIdFromToken(String token) {
        return (String) getClaimsFromToken(token).get("userId");
    }

    /**
     * 从Token中获取Claims
     */
    private Claims getClaimsFromToken(String token) {
        return Jwts.parserBuilder()
                .setSigningKey(getSignKey())
                .build()
                .parseClaimsJws(token)
                .getBody();
    }

    /**
     * 验证Token
     */
    public boolean validateToken(String token) {
        try {
            Claims claims = getClaimsFromToken(token);
            return !claims.getExpiration().before(new Date());
        } catch (Exception e) {
            return false;
        }
    }

    /**
     * 检查Token是否即将过期（1小时内）
     */
    public boolean isTokenExpiringSoon(String token) {
        try {
            Claims claims = getClaimsFromToken(token);
            Date expiration = claims.getExpiration();
            Date oneHourLater = new Date(System.currentTimeMillis() + 3600 * 1000);
            return expiration.before(oneHourLater);
        } catch (Exception e) {
            return true;
        }
    }

    /**
     * 获取Token过期时间
     */
    public Date getExpirationDateFromToken(String token) {
        return getClaimsFromToken(token).getExpiration();
    }
}
```

```java
import com.pindou.timer.common.annotation.LogOperation;
import com.pindou.timer.common.util.IpUtil;
import com.pindou.timer.modules.log.entity.OperationLog;
import com.pindou.timer.modules.log.service.LogService;
import com.alibaba.fastjson.JSON;
import lombok.extern.slf4j.Slf4j;
import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.annotation.Pointcut;
import org.aspectj.lang.reflect.MethodSignature;
import org.springframework.stereotype.Component;
import org.springframework.web.context.request.RequestContextHolder;
import org.springframework.web.context.request.ServletRequestAttributes;

import javax.servlet.http.HttpServletRequest;
import java.lang.reflect.Method;
import java.util.Date;

/**
 * 操作日志切面
 */
@Slf4j
@Aspect
@Component
public class LogAspect {

    private final LogService logService;

    public LogAspect(LogService logService) {
        this.logService = logService;
    }

    /**
     * 配置切入点
     */
    @Pointcut("@annotation(com.pindou.timer.common.annotation.LogOperation)")
    public void logPointCut() {
    }

    /**
     * 环绕通知
     */
    @Around("logPointCut()")
    public Object around(ProceedingJoinPoint point) throws Throwable {
        long beginTime = System.currentTimeMillis();
        Object result = point.proceed();
        long time = System.currentTimeMillis() - beginTime;

        // 保存日志
        saveLog(point, time, result);
        return result;
    }

    /**
     * 保存日志
     */
    private void saveLog(ProceedingJoinPoint joinPoint, long time, Object result) {
        try {
            // 获取Request
            ServletRequestAttributes attributes = (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();
            if (attributes == null) {
                return;
            }
            HttpServletRequest request = attributes.getRequest();

            // 获取注解
            MethodSignature signature = (MethodSignature) joinPoint.getSignature();
            Method method = signature.getMethod();
            LogOperation logAnnotation = method.getAnnotation(LogOperation.class);

            // 构建日志对象
            OperationLog logEntity = new OperationLog();
            logEntity.setAction(logAnnotation.action());
            logEntity.setActionName(logAnnotation.name());
            logEntity.setIp(IpUtil.getIpAddr(request));
            logEntity.setUserAgent(request.getHeader("User-Agent"));
            logEntity.setCreateTime(new Date());
            logEntity.setExecuteTime(time);

            // TODO: 从SecurityContext获取当前用户信息
            // logEntity.setUserId(currentUser.getId());
            // logEntity.setUsername(currentUser.getUsername());

            // 保存日志（异步）
            logService.saveAsync(logEntity);

        } catch (Exception e) {
            log.error("保存操作日志失败", e);
        }
    }
}
```

**请求头：**
```
Content-Type: application/json
```

**响应格式：**
```typescript
interface ApiResponse<T = any> {
  code: number;           // 状态码：200=成功 其他=失败
  message: string;        // 响应消息
  data?: T;              // 响应数据
  timestamp?: number;    // 服务器时间戳
}

interface PageResult<T> {
  list: T[];
  total: number;
  page: number;
  pageSize: number;
}
```

**状态码：**
| code | 说明 |
|------|------|
| 200 | 成功 |
| 400 | 请求参数错误 |
| 401 | 未认证/Token过期 |
| 403 | 无权限 |
| 404 | 资源不存在 |
| 409 | 资源冲突（如用户名已存在） |
| 500 | 服务器错误 |

```typescript
import type { Directive, DirectiveBinding } from 'vue';
import { useUserStore } from '@/stores/user';

/**
 * 权限指令
 * 用法：v-permission="'table:start'" 或 v-permission="['table:start', 'table:end']"
 */
export const permission: Directive = {
  mounted(el: HTMLElement, binding: DirectiveBinding<string | string[]>) {
    const { value } = binding;
    const userStore = useUserStore();

    if (value) {
      const permissions = Array.isArray(value) ? value : [value];
      const hasPermission = permissions.some(p => userStore.hasPermission(p));

      if (!hasPermission) {
        // 移除元素
        el.parentNode?.removeChild(el);
      }
    }
  }
};

/**
 * 权限禁用指令（保留元素但禁用）
 * 用法：v-permission-disabled="'table:start'"
 */
export const permissionDisabled: Directive = {
  mounted(el: HTMLElement, binding: DirectiveBinding<string>) {
    const { value } = binding;
    const userStore = useUserStore();

    if (value && !userStore.hasPermission(value)) {
      // 设置禁用状态
      if (el.tagName === 'BUTTON') {
        el.setAttribute('disabled', 'true');
        el.classList.add('is-disabled');
      } else {
        el.classList.add('is-disabled');
        el.style.pointerEvents = 'none';
        el.style.opacity = '0.6';
      }
    }
  }
};
```

### 8.2 权限计算函数

**文件路径：** `src/composables/usePermission.ts`

```typescript
import { computed } from 'vue';
import { useUserStore } from '@/stores/user';

export function usePermission() {
  const userStore = useUserStore();

  /**
   * 检查是否有权限
   */
  const hasPermission = (permission: string | string[]): boolean => {
    const permissions = Array.isArray(permission) ? permission : [permission];
    return permissions.some(p => userStore.hasPermission(p));
  };

  /**
   * 检查是否有任意权限
   */
  const hasAnyPermission = (...permissions: string[]): boolean => {
    return permissions.some(p => userStore.hasPermission(p));
  };

  /**
   * 检查是否有所有权限
   */
  const hasAllPermissions = (...permissions: string[]): boolean => {
    return permissions.every(p => userStore.hasPermission(p));
  };

  /**
   * 获取角色
   */
  const role = computed(() => userStore.userInfo?.roleName);

  /**
   * 是否是超级管理员
   */
  const isSuperAdmin = computed(() => userStore.userInfo?.roleId === 'role_super_admin');

  /**
   * 是否是管理员（超管或店长）
   */
  const isAdmin = computed(() =>
    ['role_super_admin', 'role_manager'].includes(userStore.userInfo?.roleId || '')
  );

  return {
    hasPermission,
    hasAnyPermission,
```

## 九、实时计时机制设计

### 9.1 计时逻辑

**文件路径：** `src/composables/useTableTimer.ts`

```typescript
import { ref, computed, watch, onUnmounted } from 'vue';
import type { Ref } from 'vue';
import type { Table } from '@/types/table';
import type { SystemConfig } from '@/types/config';

export interface TimerDisplay {
  formattedDuration: string;      // HH:MM:SS
  formattedAmount: string;        // ¥XX.XX
  remainSeconds: number;          // 剩余秒数
  overtimeSeconds: number;        // 超时秒数
  isExpiring: boolean;            // 是否即将到期
  isTimeout: boolean;             // 是否已超时
}

export function useTableTimer(table: Ref<Table>, config: Ref<SystemConfig>) {
  const currentTime = ref(Date.now());
  let timer: NodeJS.Timeout | null = null;

  // 格式化时长 HH:MM:SS
  const formatDuration = (seconds: number): string => {
    const h = Math.floor(seconds / 3600);
    const m = Math.floor((seconds % 3600) / 60);
    const s = seconds % 60;
    return `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}:${s.toString().padStart(2, '0')}`;
  };

  // 计算已用时长（秒）
  const usedSeconds = computed(() => {
    if (table.value.status !== 'using' && table.value.status !== 'paused') {
      return 0;
    }

    let duration = Math.floor((currentTime.value - table.value.startTime!) / 1000);

    // 减去累计暂停时长
    duration -= table.value.pauseAccumulated || 0;

    // 如果当前是暂停状态，减去本次暂停时长
    if (table.value.status === 'paused' && table.value.lastPauseTime) {
      const pausedDuration = Math.floor((currentTime.value - table.value.lastPauseTime) / 1000);
      duration -= pausedDuration;
    }

    return Math.max(0, duration);
  });

  // 计算当前金额
  const currentAmount = computed(() => {
    const seconds = usedSeconds.value;
    const { billingRule } = config.value;
    let amount = 0;

    if (billingRule.type === 'hour') {
      // 按小时计费
      amount = (seconds / 3600) * billingRule.pricePerHour;
    } else {
      // 按分钟计费
      amount = (seconds / 60) * billingRule.pricePerMinute;
    }

    // 超时计费
    if (table.value.presetDuration && seconds > table.value.presetDuration) {
      const normalAmount = (table.value.presetDuration / 3600) * billingRule.pricePerHour;
      const overtimeSeconds = seconds - table.value.presetDuration;
      const overtimeAmount = (overtimeSeconds / 3600) * billingRule.pricePerHour * billingRule.overtimeRate;
      amount = normalAmount + overtimeAmount;
    }

    return Math.max(0, amount);
  });

  // 计算剩余时间
  const remainSeconds = computed(() => {
    if (!table.value.presetDuration) return 0;
    return Math.max(0, table.value.presetDuration - usedSeconds.value);
  });

  // 计算超时时间
  const overtimeSeconds = computed(() => {
    if (!table.value.presetDuration) return 0;
    return Math.max(0, usedSeconds.value - table.value.presetDuration);
  });

  // 是否即将到期
  const isExpiring = computed(() => {
    if (!table.value.presetDuration) return false;
    const threshold = config.value.remindConfig.threshold * 60; // 转换为秒
    return remainSeconds.value > 0 && remainSeconds.value <= threshold;
  });

  // 是否已超时
  const isTimeout = computed(() => {
    if (!table.value.presetDuration) return false;
    return overtimeSeconds.value > 0;
  });

  // 显示信息
  const display = computed<TimerDisplay>(() => ({
    formattedDuration: formatDuration(usedSeconds.value),
    formattedAmount: `¥${currentAmount.value.toFixed(2)}`,
    remainSeconds: remainSeconds.value,
    overtimeSeconds: overtimeSeconds.value,
    isExpiring: isExpiring.value,
    isTimeout: isTimeout.value
  }));

  // 启动计时器
  const startTimer = () => {
    if (timer) return;

    timer = setInterval(() => {
      currentTime.value = Date.now();
    }, 1000);
  };

  // 停止计时器
  const stopTimer = () => {
    if (timer) {
      clearInterval(timer);
      timer = null;
    }
  };

  // 监听桌台状态
  watch(() => table.value.status, (newStatus) => {
    if (newStatus === 'using' || newStatus === 'paused') {
      startTimer();
    } else {
      stopTimer();
      currentTime.value = Date.now();
    }
  }, { immediate: true });

  onUnmounted(() => {
    stopTimer();
  });

  return {
    display,
    usedSeconds,
    currentAmount,
    remainSeconds,
    overtimeSeconds,
    isExpiring,
    isTimeout
  };
}
```

### 10.2 提醒检测逻辑

**文件路径：** `src/composables/useRemindChecker.ts`

```typescript
import { watch, onUnmounted } from 'vue';
import { useRemindStore } from '@/stores/remind';
import { useConfigStore } from '@/stores/config';
import { useSound } from '@/composables/useSound';
import type { Table } from '@/types/table';

export function useRemindChecker(tables: Ref<Table[]>) {
  const remindStore = useRemindStore();
  const configStore = useConfigStore();
  const { playNotify, stopRepeat } = useSound();

  let checkTimer: NodeJS.Timeout | null = null;

  // 检查单个桌台
  const checkTable = (table: Table) => {
    if (table.status !== 'using' || !table.presetDuration) {
      return null;
    }

    const now = Date.now();
    const usedSeconds = Math.floor((now - table.startTime!) / 1000) - (table.pauseAccumulated || 0);
    const remainSeconds = table.presetDuration - usedSeconds;

    // 即将到期检查
    if (remainSeconds > 0 && remainSeconds <= configStore.remindConfig.threshold * 60) {
      return {
        tableId: table.id,
        tableName: table.name,
        type: 'expiring' as const,
        remainSeconds,
        overtimeSeconds: 0,
        timestamp: now
      };
    }

    // 超时检查
    if (remainSeconds < 0) {
      return {
        tableId: table.id,
        tableName: table.name,
        type: 'timeout' as const,
        remainSeconds: 0,
        overtimeSeconds: Math.abs(remainSeconds),
        timestamp: now
      };
    }

    return null;
  };

  // 执行检查
  const doCheck = () => {
    const newReminders: Reminder[] = [];

    for (const table of tables.value) {
      const reminder = checkTable(table);
      if (reminder) {
        newReminders.push(reminder);
      }
    }

    // 更新提醒列表
    remindStore.clearReminders();
    for (const reminder of newReminders) {
      remindStore.addReminder(reminder);

      // 播放声音
      if (configStore.remindConfig.soundEnabled) {
        const isRepeat = reminder.type === 'timeout';
        playNotify(reminder.type, isRepeat);
      }
    }

    // 如果没有超时提醒，停止重复播放
    const hasTimeout = newReminders.some(r => r.type === 'timeout');
    if (!hasTimeout) {
      stopRepeat();
    }
  };

  // 启动检查定时器
  const startCheck = () => {
    // 立即执行一次
    doCheck();

    // 每5秒检查一次
    checkTimer = setInterval(doCheck, 5000);
  };

  // 停止检查
  const stopCheck = () => {
    if (checkTimer) {
      clearInterval(checkTimer);
      checkTimer = null;
    }
    stopRepeat();
  };

  // 监听桌台列表变化
  watch(tables, () => {
    doCheck();
  }, { deep: true });

  // 清理
  onUnmounted(() => {
    stopCheck();
  });

  return {
    startCheck,
    stopCheck
  };
}
```

```vue
      };
```

```typescript
// 响应拦截器

  (response) => {
    const { code, message, data } = response.data;

    // 成功响应
    if (code === 200) {
      return { ...response.data, data };
    }

    // 业务错误
    ElMessage.error(message || '请求失败');
    return Promise.reject(new Error(message || '请求失败'));
  },
  (error: AxiosError) => {
    const { response } = error;

    if (response) {
      switch (response.status) {
        case 401:
          // Token过期或无效
          ElMessage.error('登录已过期，请重新登录');
          const userStore = useUserStore();
          userStore.logout();
          router.push('/login');
          break;
        case 403:
          ElMessage.error('没有权限访问');
          break;
        case 404:
          ElMessage.error('请求的资源不存在');
          break;
        case 500:
          ElMessage.error('服务器错误');
          break;
        default:
          ElMessage.error('请求失败');
      }
    } else if (error.code === 'ECONNABORTED') {
      ElMessage.error('请求超时');
    } else {
      ElMessage.error('网络错误');
    }

    return Promise.reject(error);
```

```typescript
/**
 * 防抖
 */
export function debounce<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let timer: NodeJS.Timeout | null = null;

  return function (this: any, ...args: Parameters<T>) {
    if (timer) clearTimeout(timer);

    timer = setTimeout(() => {
      fn.apply(this, args);
    }, delay);
  };
}

/**
 * 节流
 */
export function throttle<T extends (...args: any[]) => any>(
  fn: T,
  delay: number
): (...args: Parameters<T>) => void {
  let lastTime = 0;

  return function (this: any, ...args: Parameters<T>) {
    const now = Date.now();

    if (now - lastTime >= delay) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
}
```

```typescript
export async function cachedRequest<T>(
  key: string,
  fn: () => Promise<T>,
  ttl: number = 60000
```
